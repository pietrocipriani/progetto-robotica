56void interpolation3(
57 const Params<Point>& start,
58 const Params<Point>& via,
59 const Params<Point>& end,
60 typename SequencedFunction<TimeFunction<Point>, Time>::Container& chain
61) {
62 using namespace uniformed_rotation_algebra;
63
64 // The chain must start with a Mode::Point.
65 assert(start_mode == Mode::Point || !chain.empty());
66
67 // Timestamps.
68 auto t12 = QuadraticTimestamps::build<start_mode, Mode::ViaPoint>(start, via);
69 auto t23 = QuadraticTimestamps::build<Mode::ViaPoint, end_mode>(via, end);
70
71 // Velocities.
72 auto v12 = unlazy((via.point - start.point) / t12.delta);
73 auto v23 = unlazy((end.point - via.point) / t23.delta);
74
75 if constexpr (start_mode == Mode::Point) {
76 auto q = quadratic_acceleration(start.point, v12, t12.start, start.times.accel_delta);
77 assert((start.point - q(t12.start)).norm() < dummy_precision);
78 chain.emplace_back(std::move(q), t12.start);
79
80 if (t12.start_linear < t12.end_linear) {
81 auto l = linear_interpolation(start.point, via.point, t12.start + start.times.accel_delta / 2, t12.delta);
82 assert((l(t12.start_linear) - chain.back()(t12.start_linear)).norm() < dummy_precision);
83 chain.emplace_back(std::move(l), t12.start_linear);
84 }
85 }
86
87 auto q = quadratic_interpolation(chain.back()(t12.end_linear), v12, v23, t12.end_linear, via.times.accel_delta);
88 assert((q(t12.end_linear) - chain.back()(t12.end_linear)).norm() < dummy_precision);
89 chain.emplace_back(std::move(q), t12.end_linear);
90
91 if (t23.start_linear < t23.end_linear) {
92 auto l = linear_interpolation(via.point, end.point, t23.start, t23.delta);
93 assert((chain.back()(t23.start_linear) - l(t23.start_linear)).norm() < dummy_precision);
94 chain.emplace_back(std::move(l), t23.start_linear);
95 }
96
97 if constexpr (end_mode == Mode::Point) {
98 auto q = quadratic_deceleration(end.point, v23, t23.end, end.times.accel_delta);
99
100 assert((q(t23.end_linear) - chain.back()(t23.end_linear)).norm() < dummy_precision);
101 assert((q(t23.end) - end.point).norm() < dummy_precision);
102
103 chain.emplace_back(std::move(q), t23.end_linear);
104 }

117template<class Point, template<class T> class Container = std::initializer_list>
118auto parabolic_interpolation(
119 const Params<Point>& start,
120 const Container<Params<Point>>& via_points,
121 const Params<Point>& end
122) {
123 using namespace internal;
124 using Sequence = SequencedFunction<TimeFunction<Point>, Time>;
125
126 if (via_points.size() == 0) {
127 return stop_and_play_interpolation(start, {end});
128 }
129
130 // The resulting function is a chain of interpolations.
131 typename Sequence::Container chain;
132
133 // - quadratic acceleration from the starting node.
134 // - linear movement after the starting node.
135 // - quadratic deceleration to the final node.
136 // - For each via point:
137 // - quadratic acceleration.
138 // - linear movement.
139 chain.reserve(3 + 2 * via_points.size());
140
141 if (via_points.size() == 1) {
142 interpolation3<Mode::Point, Mode::Point>(start, *via_points.begin(), end, chain);
143 } else {
144 auto i1 = via_points.begin();
145 // This is a valid iterator as via_points.size() >= 2.
146 auto i2 = std::next(i1);
147
148 interpolation3<Mode::Point, Mode::ViaPoint>(start, *i1, *i2, chain);
149 for (auto i3 = std::next(i2); i3 != via_points.end(); i1 = i2, i2 = i3, i3 = std::next(i2)) {
150 interpolation3<Mode::ViaPoint, Mode::ViaPoint>(*i1, *i2, *i3, chain);
151 }
152 interpolation3<Mode::ViaPoint, Mode::Point>(*i1, *i2, end, chain);
153 }
154
155 return Sequence(std::move(chain));
156}
