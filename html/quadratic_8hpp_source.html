32template<class Point, class Velocity>
33TimeFunction<Point> _quadratic_interpolation(
34 const Point& initial_position,
35 const Velocity& initial_velocity,
36 const Velocity& final_velocity,
37 const Time& start_time,
38 const Time& duration
39) {
40 using namespace uniformed_rotation_algebra;
41
42 // 1/2 a. Due to strict policies in the manipulability of js/os::Position.
43 auto acceleration2 = unlazy((final_velocity - initial_velocity) / (2 * duration));
44
45 return [=, acc = std::move(acceleration2)](const Time& time) {
46 auto t = time - start_time;
47 return initial_position + initial_velocity * t + acc * t * t;
48 };
