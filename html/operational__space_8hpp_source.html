135 const Base vector() const {
136 Base result;

138 // The linear part is just the linear part.
139 auto builder = result << _linear;
140
141 if constexpr (angular_system == coord::Lie) {
142 // For the angular part in lie algebra we consider the angular pseudo-vector.
143 // NOTE: this implementation is in reality only valid for quaternions.
144
145 auto sin_theta_2 = _angular.vec().norm();
146 if (sin_theta_2 < dummy_precision) {
147 // To avoid numeric approximation bad conditioning.
148 builder , _angular.vec() * 2;
149 } else {
150 // Extraction of the theta angle.
151 auto omega = 2 * std::asin(sin_theta_2);
152 builder , _angular.vec().normalized() * omega;
153 }
154 } else {
155 // In euler algebra the angular part is already represented as a vector.
156 builder , _linear;
157 }

159 builder.finished();
