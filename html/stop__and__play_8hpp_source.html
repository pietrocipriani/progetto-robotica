83
84 // The resulting function is a chain of interpolations.
85 typename Sequence::Container chain;
86
87
88 // points.size() - 1 segments.
89 // For each segment 3 functions are needed (acc + lin + dec).
90 chain.reserve(3 * (points.size() - 1));
91
92
93 if (points.size() == 0) {
94 // Constant function in case of only one point.
95 chain.emplace_back([p = start.point]([[maybe_unused]] const Time& _) {return p;}, start.times.time);
96 } else {
97 // This is a valid iterator as points.size() >= 1.
98 auto i1 = points.begin();
99
100 interpolation2(start, *i1, chain);
101 for (; std::next(i1) != points.end(); i1 = std::next(i1)) {
102 interpolation2(*i1, *std::next(i1), chain);
103 }
104 }
105
106 auto s = Sequence(std::move(chain));
